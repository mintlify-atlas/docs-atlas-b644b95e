## Overview

The `setValues` function can return either a static array or an RxJS Observable. This enables dynamic data loading from APIs, databases, or any asynchronous source.

## HTTP API Example

<CodeGroup>

```typescript component.ts
import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { map, catchError } from 'rxjs/operators';
import { of } from 'rxjs';
import { NgxDependDropdownService } from 'ngx-depend-dropdown';
import { DropdownConfig, DropdownData } from 'ngx-depend-dropdown';

@Component({
  selector: 'app-observable-example',
  templateUrl: './observable-example.component.html',
  styleUrls: ['./observable-example.component.css']
})
export class ObservableExampleComponent implements OnInit {
  dropdownData: DropdownData = [];
  dropdownConfig: DropdownConfig = {};
  loading = false;

  constructor(
    private http: HttpClient,
    private dropdownService: NgxDependDropdownService
  ) {}

  ngOnInit(): void {
    this.setupObservableDropdowns();
    this.dropdownConfig = {
      unavailableType: 'disabled',
      containerClass: 'dropdown-wrapper',
      nodeClass: 'custom-select'
    };
  }

  setupObservableDropdowns() {
    this.dropdownData = [
      {
        name: 'user',
        visibility: () => true,
        selectedValue: null,
        placeholder: 'Select a user',
        label: 'User',
        values: [],
        setValues: () => {
          return this.http
            .get('https://jsonplaceholder.typicode.com/users')
            .pipe(
              map((data: any) => {
                return data.map((item: any) => ({
                  id: item.id,
                  name: item.name,
                  email: item.email,
                  username: item.username
                }));
              }),
              catchError(error => {
                console.error('Error loading users:', error);
                return of([]);
              })
            );
        }
      },
      {
        name: 'post',
        visibility: () => {
          return !!this.dropdownService.getNodeValue('user');
        },
        selectedValue: null,
        placeholder: 'Select a post',
        label: 'Posts by User',
        values: [],
        dependent: 'user',
        setValues: () => {
          const user = this.dropdownService.getNodeValue('user');
          
          return this.http
            .get(`https://jsonplaceholder.typicode.com/posts?userId=${user.id}`)
            .pipe(
              map((data: any) => {
                return data.map((item: any) => ({
                  id: item.id,
                  name: item.title,
                  body: item.body
                }));
              }),
              catchError(error => {
                console.error('Error loading posts:', error);
                return of([]);
              })
            );
        }
      },
      {
        name: 'comment',
        visibility: () => {
          return !!this.dropdownService.getNodeValue('post');
        },
        selectedValue: null,
        placeholder: 'Select a comment',
        label: 'Comments on Post',
        values: [],
        dependent: 'post',
        setValues: () => {
          const post = this.dropdownService.getNodeValue('post');
          
          return this.http
            .get(`https://jsonplaceholder.typicode.com/comments?postId=${post.id}`)
            .pipe(
              map((data: any) => {
                return data.map((item: any) => ({
                  id: item.id,
                  name: `${item.name} (${item.email})`,
                  body: item.body
                }));
              }),
              catchError(error => {
                console.error('Error loading comments:', error);
                return of([]);
              })
            );
        }
      }
    ];
  }

  handleChange(event: any) {
    console.log('Selection changed:', event);
    this.loading = true;
    
    // Simulate loading state
    setTimeout(() => {
      this.loading = false;
    }, 500);
  }
}
```

## Module Configuration

<Note>
  Don't forget to import `HttpClientModule` to enable HTTP requests in your application.
</Note>

```typescript app.module.ts
import { NgModule } from '@angular/core';
import { BrowserModule } from '@angular/platform-browser';
import { HttpClientModule } from '@angular/common/http';
import { NgxDependDropdownModule } from 'ngx-depend-dropdown';
import { ObservableExampleComponent } from './observable-example.component';

@NgModule({
  declarations: [ObservableExampleComponent],
  imports: [
    BrowserModule,
    HttpClientModule,
    NgxDependDropdownModule
  ],
  bootstrap: [ObservableExampleComponent]
})
export class AppModule { }
```

## Using a Service for Data

<Tip>
  For better code organization, create a dedicated service to handle API calls.
</Tip>

<CodeGroup>

```typescript data.service.ts
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { map, catchError } from 'rxjs/operators';
import { of } from 'rxjs';

@Injectable({
  providedIn: 'root'
})
export class DataService {
  private apiUrl = 'https://api.example.com';

  constructor(private http: HttpClient) {}

  getCountries(): Observable<any[]> {
    return this.http.get<any[]>(`${this.apiUrl}/countries`).pipe(
      map(data => data.map(item => ({
        id: item.code,
        name: item.name
      }))),
      catchError(error => {
        console.error('Error fetching countries:', error);
        return of([]);
      })
    );
  }

  getStatesByCountry(countryId: string): Observable<any[]> {
    return this.http.get<any[]>(`${this.apiUrl}/countries/${countryId}/states`).pipe(
      map(data => data.map(item => ({
        id: item.code,
        name: item.name
      }))),
      catchError(error => {
        console.error('Error fetching states:', error);
        return of([]);
      })
    );
  }
}
```

```typescript component-with-service.ts
import { Component, OnInit } from '@angular/core';
import { DataService } from './data.service';
import { NgxDependDropdownService } from 'ngx-depend-dropdown';
import { DropdownData } from 'ngx-depend-dropdown';

@Component({
  selector: 'app-service-example',
  template: `
    <ngx-depend-dropdown 
      [data]="dropdownData" 
      [config]="dropdownConfig">
    </ngx-depend-dropdown>
  `
})
export class ServiceExampleComponent implements OnInit {
  dropdownData: DropdownData = [];
  dropdownConfig = { unavailableType: 'disabled' };

  constructor(
    private dataService: DataService,
    private dropdownService: NgxDependDropdownService
  ) {}

  ngOnInit(): void {
    this.dropdownData = [
      {
        name: 'country',
        visibility: () => true,
        selectedValue: null,
        placeholder: 'Select country',
        values: [],
        setValues: () => this.dataService.getCountries()
      },
      {
        name: 'state',
        visibility: () => !!this.dropdownService.getNodeValue('country'),
        selectedValue: null,
        placeholder: 'Select state',
        values: [],
        dependent: 'country',
        setValues: () => {
          const country = this.dropdownService.getNodeValue('country');
          return this.dataService.getStatesByCountry(country.id);
        }
      }
    ];
  }
}
```

## Error Handling

<Note>
  Always use `catchError` to handle API failures gracefully and return an empty array to prevent dropdown errors.
</Note>

```typescript
setValues: () => {
  return this.http.get('https://api.example.com/data').pipe(
    map((data: any) => data.map(item => ({
      id: item.id,
      name: item.name
    }))),
    catchError(error => {
      console.error('API Error:', error);
      // Return empty array on error
      return of([]);
    })
  );
}
```

## Loading States

You can track loading states by monitoring the onChange event:

```typescript
handleChange(event: any) {
  this.loading = true;
  
  // The library automatically handles the async loading
  // Set loading to false after a short delay or in the subscription
  setTimeout(() => {
    this.loading = false;
  }, 500);
}
```
