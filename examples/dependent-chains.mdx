## Overview

Dependent chains allow you to create dropdowns where each selection determines the options available in the next dropdown. This is perfect for hierarchical data like Country → State → City.

## Three-Level Dependent Dropdown

<CodeGroup>

```typescript component.ts
import { Component, OnInit } from '@angular/core';
import { NgxDependDropdownService } from 'ngx-depend-dropdown';
import { DropdownConfig, DropdownData } from 'ngx-depend-dropdown';

@Component({
  selector: 'app-dependent-example',
  templateUrl: './dependent-example.component.html',
  styleUrls: ['./dependent-example.component.css']
})
export class DependentExampleComponent implements OnInit {
  dropdownData: DropdownData = [];
  dropdownConfig: DropdownConfig = {};

  constructor(private dropdownService: NgxDependDropdownService) {}

  ngOnInit(): void {
    this.setupDependentDropdowns();
    this.dropdownConfig = {
      unavailableType: 'disabled',
      containerClass: ['row', 'g-3'],
      nodeClass: ['form-control', 'form-select']
    };
  }

  setupDependentDropdowns() {
    this.dropdownData = [
      {
        name: 'country',
        visibility: () => true,
        selectedValue: null,
        placeholder: 'Select a country',
        label: 'Country',
        values: [],
        setValues: () => {
          return [
            { id: 1, name: 'United States' },
            { id: 2, name: 'Canada' },
            { id: 3, name: 'United Kingdom' }
          ];
        }
      },
      {
        name: 'state',
        visibility: () => {
          return !!this.dropdownService.getNodeValue('country');
        },
        selectedValue: null,
        placeholder: 'Select a state',
        label: 'State / Province',
        values: [],
        dependent: 'country',
        setValues: () => {
          const country = this.dropdownService.getNodeValue('country');
          
          if (country?.id === 1) {
            return [
              { id: 101, name: 'California' },
              { id: 102, name: 'Texas' },
              { id: 103, name: 'New York' },
              { id: 104, name: 'Florida' }
            ];
          } else if (country?.id === 2) {
            return [
              { id: 201, name: 'Ontario' },
              { id: 202, name: 'Quebec' },
              { id: 203, name: 'British Columbia' }
            ];
          } else if (country?.id === 3) {
            return [
              { id: 301, name: 'England' },
              { id: 302, name: 'Scotland' },
              { id: 303, name: 'Wales' }
            ];
          }
          return [];
        }
      },
      {
        name: 'city',
        visibility: () => {
          return !!this.dropdownService.getNodeValue('state');
        },
        selectedValue: null,
        placeholder: 'Select a city',
        label: 'City',
        values: [],
        dependent: 'state',
        setValues: () => {
          const state = this.dropdownService.getNodeValue('state');
          
          // California cities
          if (state?.id === 101) {
            return [
              { id: 1001, name: 'Los Angeles' },
              { id: 1002, name: 'San Francisco' },
              { id: 1003, name: 'San Diego' }
            ];
          }
          // Texas cities
          else if (state?.id === 102) {
            return [
              { id: 1004, name: 'Houston' },
              { id: 1005, name: 'Dallas' },
              { id: 1006, name: 'Austin' }
            ];
          }
          // Ontario cities
          else if (state?.id === 201) {
            return [
              { id: 2001, name: 'Toronto' },
              { id: 2002, name: 'Ottawa' },
              { id: 2003, name: 'Mississauga' }
            ];
          }
          // England cities
          else if (state?.id === 301) {
            return [
              { id: 3001, name: 'London' },
              { id: 3002, name: 'Manchester' },
              { id: 3003, name: 'Birmingham' }
            ];
          }
          
          return [];
        }
      }
    ];
  }

  handleChange(event: any) {
    console.log('Changed dropdown:', event.changedDropdown.name);
    console.log('Selected value:', event.changedDropdown.selectedValue);
  }
}
```

## How It Works

<Note>
  The `dependent` property creates the relationship between dropdowns. When the parent dropdown changes, dependent children are automatically reset.
</Note>

### Key Features Explained

1. **Visibility Control**: Each dependent dropdown uses the `visibility` function to check if its parent has a value:
   ```typescript
   visibility: () => {
     return !!this.dropdownService.getNodeValue('country');
   }
   ```

2. **Dependent Property**: Links the dropdown to its parent:
   ```typescript
   dependent: 'country'
   ```

3. **Dynamic Values**: The `setValues` function uses the parent's selected value to determine options:
   ```typescript
   setValues: () => {
     const country = this.dropdownService.getNodeValue('country');
     // Return different options based on country
   }
   ```

## Service Methods

<Tip>
  Use the `NgxDependDropdownService` to access dropdown data from anywhere in your component.
</Tip>

### Available Methods

- **getNodeValue(nodeName: string)**: Returns the selected value of a specific dropdown
- **getNode(nodeName: string)**: Returns the complete node data for a dropdown

```typescript
// Example usage
const selectedCountry = this.dropdownService.getNodeValue('country');
const countryNode = this.dropdownService.getNode('country');
```

## Multiple Dependencies

You can also create a dropdown that depends on multiple parents:

```typescript
{
  name: 'shipping-option',
  visibility: () => {
    return !!this.dropdownService.getNodeValue('country') && 
           !!this.dropdownService.getNodeValue('state');
  },
  dependent: ['country', 'state'],
  setValues: () => {
    // Returns options based on both country and state
  }
}
```
