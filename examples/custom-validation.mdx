## Overview

The ngx-depend-dropdown library supports built-in validation with customizable error messages. You can mark dropdowns as required and display custom error messages when no value is selected.

## Basic Validation Example

<CodeGroup>

```typescript component.ts
import { Component, OnInit } from '@angular/core';
import { NgxDependDropdownService } from 'ngx-depend-dropdown';
import { DropdownConfig, DropdownData } from 'ngx-depend-dropdown';

@Component({
  selector: 'app-validation-example',
  templateUrl: './validation-example.component.html',
  styleUrls: ['./validation-example.component.css']
})
export class ValidationExampleComponent implements OnInit {
  dropdownData: DropdownData = [];
  dropdownConfig: DropdownConfig = {};
  formSubmitted = false;

  constructor(private dropdownService: NgxDependDropdownService) {}

  ngOnInit(): void {
    this.setupValidationDropdowns();
    this.dropdownConfig = {
      unavailableType: 'disabled',
      containerClass: 'form-group',
      nodeClass: 'form-control'
    };
  }

  setupValidationDropdowns() {
    this.dropdownData = [
      {
        name: 'department',
        visibility: () => true,
        selectedValue: null,
        placeholder: 'Choose your department',
        label: 'Department',
        values: [],
        required: true,
        errorMessage: 'Please select a department to continue',
        setValues: () => {
          return [
            { id: 1, name: 'Engineering' },
            { id: 2, name: 'Marketing' },
            { id: 3, name: 'Sales' },
            { id: 4, name: 'Human Resources' }
          ];
        }
      },
      {
        name: 'team',
        visibility: () => {
          return !!this.dropdownService.getNodeValue('department');
        },
        selectedValue: null,
        placeholder: 'Choose your team',
        label: 'Team',
        values: [],
        dependent: 'department',
        required: true,
        errorMessage: 'Team selection is mandatory',
        setValues: () => {
          const dept = this.dropdownService.getNodeValue('department');
          
          if (dept?.id === 1) {
            return [
              { id: 101, name: 'Frontend Team' },
              { id: 102, name: 'Backend Team' },
              { id: 103, name: 'DevOps Team' }
            ];
          } else if (dept?.id === 2) {
            return [
              { id: 201, name: 'Content Team' },
              { id: 202, name: 'Social Media Team' },
              { id: 203, name: 'SEO Team' }
            ];
          } else if (dept?.id === 3) {
            return [
              { id: 301, name: 'Inside Sales' },
              { id: 302, name: 'Field Sales' },
              { id: 303, name: 'Account Management' }
            ];
          } else if (dept?.id === 4) {
            return [
              { id: 401, name: 'Recruitment' },
              { id: 402, name: 'Payroll' },
              { id: 403, name: 'Employee Relations' }
            ];
          }
          return [];
        }
      },
      {
        name: 'role',
        visibility: () => {
          return !!this.dropdownService.getNodeValue('team');
        },
        selectedValue: null,
        placeholder: 'Choose your role',
        label: 'Role',
        values: [],
        dependent: 'team',
        required: true,
        errorMessage: 'You must specify your role in the team',
        setValues: () => {
          return [
            { id: 1, name: 'Team Lead' },
            { id: 2, name: 'Senior Member' },
            { id: 3, name: 'Member' },
            { id: 4, name: 'Intern' }
          ];
        }
      }
    ];
  }

  isFormValid(): boolean {
    return this.dropdownData.every(node => {
      if (node.required) {
        return node.selectedValue !== null && node.selectedValue !== undefined;
      }
      return true;
    });
  }

  submitForm() {
    this.formSubmitted = true;
    
    if (this.isFormValid()) {
      console.log('Form is valid!');
      console.log('Department:', this.dropdownData[0].selectedValue);
      console.log('Team:', this.dropdownData[1].selectedValue);
      console.log('Role:', this.dropdownData[2].selectedValue);
      
      alert('Form submitted successfully!');
    } else {
      console.log('Form has validation errors');
      alert('Please complete all required fields');
    }
  }

  resetForm() {
    this.formSubmitted = false;
    this.dropdownData.forEach(node => {
      node.selectedValue = null;
    });
  }

  handleChange(event: any) {
    console.log('Dropdown changed:', event.changedDropdown.name);
  }
}
```

## Validation Properties

<Note>
  The library automatically displays error messages when a required field is not selected.
</Note>

### Required Field Properties

- **required**: Set to `true` or a string value to mark the dropdown as required
- **errorMessage**: Custom message displayed when the required field is empty

```typescript
{
  name: 'country',
  required: true,
  errorMessage: 'Please select a country',
  // ... other properties
}
```

## Conditional Validation

You can make validation conditional based on other selections:

<CodeGroup>

```typescript conditional-validation.ts
setupConditionalValidation() {
  this.dropdownData = [
    {
      name: 'subscription-type',
      visibility: () => true,
      selectedValue: null,
      placeholder: 'Select subscription type',
      label: 'Subscription Type',
      values: [],
      required: true,
      errorMessage: 'Subscription type is required',
      setValues: () => [
        { id: 1, name: 'Free' },
        { id: 2, name: 'Premium' },
        { id: 3, name: 'Enterprise' }
      ]
    },
    {
      name: 'billing-cycle',
      visibility: () => {
        const subscription = this.dropdownService.getNodeValue('subscription-type');
        return subscription?.id !== 1; // Only show for paid plans
      },
      selectedValue: null,
      placeholder: 'Select billing cycle',
      label: 'Billing Cycle',
      values: [],
      dependent: 'subscription-type',
      // Only required for paid plans
      required: () => {
        const subscription = this.dropdownService.getNodeValue('subscription-type');
        return subscription?.id !== 1;
      },
      errorMessage: 'Please select a billing cycle for your subscription',
      setValues: () => [
        { id: 1, name: 'Monthly' },
        { id: 2, name: 'Yearly' }
      ]
    }
  ];
}
```

</CodeGroup>

<Tip>
  The `required` property can accept a boolean value or a function that returns a boolean for dynamic validation.
</Tip>

## Form Validation Helper

Create a reusable validation helper method:

```typescript
isFormValid(): boolean {
  return this.dropdownData.every(node => {
    // Skip non-visible dropdowns
    if (!node.visibilityValue) {
      return true;
    }
    
    // Check if required field has a value
    if (node.required) {
      return node.selectedValue !== null && node.selectedValue !== undefined;
    }
    
    return true;
  });
}

getValidationErrors(): string[] {
  const errors: string[] = [];
  
  this.dropdownData.forEach(node => {
    if (node.required && node.visibilityValue && !node.selectedValue) {
      errors.push(node.errorMessage || `${node.label || node.name} is required`);
    }
  });
  
  return errors;
}
```

## Integration with Angular Forms

While ngx-depend-dropdown doesn't directly integrate with Angular Reactive Forms, you can sync the values:

```typescript
import { FormBuilder, FormGroup, Validators } from '@angular/forms';

export class FormIntegrationComponent implements OnInit {
  form: FormGroup;
  dropdownData: DropdownData = [];

  constructor(
    private fb: FormBuilder,
    private dropdownService: NgxDependDropdownService
  ) {
    this.form = this.fb.group({
      department: [null, Validators.required],
      team: [null, Validators.required],
      role: [null, Validators.required]
    });
  }

  handleChange(event: any) {
    const name = event.changedDropdown.name;
    const value = event.changedDropdown.selectedValue;
    
    // Update form control
    if (this.form.contains(name)) {
      this.form.patchValue({ [name]: value });
    }
  }

  submitForm() {
    if (this.form.valid) {
      console.log('Form data:', this.form.value);
    }
  }
}
```
